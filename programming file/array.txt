#include<stdio.h>
int main()
{
    int i[5],sum=0,a;
    printf("enter five numbers:");
    scanf("%d %d %d %d %d",&i[0],&i[1],&i[2],&i[3],&i[4]);
    for(a=0;a<5;a++)
        sum=sum+i[a];

    printf("the sum is : %d\n",sum);
    printf("the average is: %.2f\n",(float)sum/5);

    return 0;
}
