#include <stdio.h>

void sum(int a, int b){
	int result = a + b;
	printf("sum result %d",result);
}

void sub(int a, int b){
	int result = a - b;
	printf("\n sub result %d",result);
}

int main()
{
		int number1, number2;

		printf("Please enter your first number: ");
		scanf("%d",&number1);

		printf("Please enter your second number: ");
		scanf("%d",&number2);

		sum(number1, number2);
		sub(number1, number2);
		getchar();

		return 0;
}
